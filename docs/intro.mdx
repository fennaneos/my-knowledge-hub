<script
  dangerouslySetInnerHTML={{
    __html: `(function () {
      function hideIntroStars() {
        try {
          // Match common hrefs that point to the intro doc
          var selectors = [
            'a.menu__link[href="/"]',
            'a.menu__link[href="/docs/"]',
            'a.menu__link[href="/docs"]',
            'a.menu__link[href*="/docs/intro"]'
          ];
          var links = document.querySelectorAll(selectors.join(','));
          links.forEach(function (a) {
            // Our sidebar star wrapper is the immediate container <div> wrapping the link
            var wrapper = a.closest('div');
            if (!wrapper) return;
            var stars = wrapper.querySelector('.lux-stars');
            if (stars) stars.style.display = 'none';
            var gauge = wrapper.querySelector('.lux-gauge');
            if (gauge) gauge.style.display = 'none';
          });
        } catch (e) { /* noop */ }
      }
      if (document.readyState !== 'loading') hideIntroStars();
      else document.addEventListener('DOMContentLoaded', hideIntroStars);
      // Also run after client-side nav
      document.addEventListener('docusaurus.routeDidUpdate', hideIntroStars);
    })();`,
  }}
/>
